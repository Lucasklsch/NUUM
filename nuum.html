function showImpactAtEnemy(enemy) { // Calcula a posição do inimigo na tela let dx = enemy.x - playerX; let dy = enemy.y - playerY; let distance = Math.sqrt(dx * dx + dy * dy); let enemyAngle = Math.atan2(dx, dy) - playerA; let enemyPos = (enemyAngle + FOV / 2.0) / FOV; if (enemyPos < 0 || enemyPos > 1) return; // Define dimensões aproximadas do inimigo na tela let enemyHeight = Math.floor(screenHeight / distance); let enemyWidth = enemyHeight; let enemyColumn = Math.floor(enemyPos * screenWidth) - Math.floor(enemyWidth / 2); let enemyRow = Math.floor((screenHeight / 2) - enemyHeight / 2); // Obtém informações de posicionamento da área de renderização (screen) let screenEl = document.getElementById("screen"); let rect = screenEl.getBoundingClientRect(); let charWidth = rect.width / screenWidth; let charHeight = rect.height / screenHeight; // Calcula a posição exata para centralizar a imagem de impacto let impactX = rect.left + (enemyColumn + enemyWidth / 2) * charWidth; let impactY = rect.top + (enemyRow + enemyHeight / 2) * charHeight; // Cria dinamicamente um novo elemento para o impacto let impactEl = document.createElement("img"); impactEl.src = "impact.png"; impactEl.alt = "Impacto"; impactEl.style.position = "absolute"; impactEl.style.left = impactX + "px"; impactEl.style.top = impactY + "px"; impactEl.style.transform = "translate(-50%, -50%)"; impactEl.style.width = "50px"; // ajuste conforme necessário impactEl.style.zIndex = "3"; impactEl.style.pointerEvents = "none"; // Adiciona o elemento ao container do jogo document.getElementById("game-container").appendChild(impactEl); // Remove a imagem após 200ms (ou o tempo que achar adequado) setTimeout(() => { impactEl.remove(); }, 200); }
